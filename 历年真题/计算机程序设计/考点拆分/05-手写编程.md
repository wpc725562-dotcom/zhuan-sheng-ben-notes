## 2024 三道编程（必须默写）

### A. 3×3 按行填 201–209
```c
#include <stdio.h>
int main() {
    int a[3][3], num = 201;
    for (int i = 0; i < 3; i++)
        for (int j = 0; j < 3; j++)
            a[i][j] = num++;
    for (int i = 0; i < 3; i++) {
        for (int j = 0; j < 3; j++)
            printf("%d ", a[i][j]);
        printf("\n");
    }
    return 0;
}
```

### B. 30 个成绩找最高分
```c
#include <stdio.h>
int main() {
    int scores[30], max;
    for (int i = 0; i < 30; i++)
        scanf("%d", &scores[i]);
    max = scores[0];
    for (int i = 1; i < 30; i++)
        if (scores[i] > max) max = scores[i];
    printf("最高分 = %d\n", max);
    return 0;
}
```

### C. 1–300 奇偶分别求和
```c
#include <stdio.h>
int main() {
    int sum_odd = 0, sum_even = 0;
    for (int i = 1; i <= 300; i++) {
        if (i % 2) sum_odd += i;
        else sum_even += i;
    }
    printf("奇数和 = %d\n", sum_odd);
    printf("偶数和 = %d\n", sum_even);
    return 0;
}
```
参考：奇数和 22500，偶数和 22650。
